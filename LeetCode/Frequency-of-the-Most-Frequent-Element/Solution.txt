1class Solution {
2public:
3    int maxFrequency(vector<int>& nums, int k) {
4        sort(nums.begin(),nums.end());
5        long long int sum=0, l=0, r=0, maxFreq=0, n=nums.size();
6        for(r=0; r<n; r++){
7            sum+=nums[r];
8            while(l+1 <= r && (nums[r]*(r-l+1)>k+sum)){
9                sum-=nums[l];
10                l++;
11            }
12            maxFreq=max(maxFreq, r-l+1);
13        }
14        return maxFreq;
15    }
16};
